```rust
/// Move cursor by `delta` bytes (can be negative), clamp to file bounds, clear selection, auto-scroll.
fn move_cursor(&mut self, delta: isize) {
    let Some(file) = &self.file else { return };
    let max = file.len().saturating_sub(1);
    if delta < 0 {
        self.cursor_offset = self.cursor_offset.saturating_sub(delta.unsigned_abs());
    } else {
        self.cursor_offset = (self.cursor_offset + delta as usize).min(max);
    }
    self.selection = None;
    self.scroll_to_cursor();
}

/// Set cursor to absolute offset, clamp to file bounds, clear selection, auto-scroll.
fn set_cursor_abs(&mut self, offset: usize) {
    let Some(file) = &self.file else { return };
    self.cursor_offset = offset.min(file.len().saturating_sub(1));
    self.selection = None;
    self.scroll_to_cursor();
}
```

```rust
/// Move cursor by delta, extending selection. Anchor is set on first shift-move.
fn move_cursor_select(&mut self, delta: isize) {
    let Some(file) = &self.file else { return };
    let max = file.len().saturating_sub(1);
    let anchor = self.selection_anchor.unwrap_or(self.cursor_offset);
    self.selection_anchor = Some(anchor);

    if delta < 0 {
        self.cursor_offset = self.cursor_offset.saturating_sub(delta.unsigned_abs());
    } else {
        self.cursor_offset = (self.cursor_offset + delta as usize).min(max);
    }
    self.selection = Some(Selection::new(anchor, self.cursor_offset));
    self.scroll_to_cursor();
}

fn set_cursor_select(&mut self, offset: usize) {
    let Some(file) = &self.file else { return };
    let anchor = self.selection_anchor.unwrap_or(self.cursor_offset);
    self.selection_anchor = Some(anchor);
    self.cursor_offset = offset.min(file.len().saturating_sub(1));
    self.selection = Some(Selection::new(anchor, self.cursor_offset));
    self.scroll_to_cursor();
}
```

```rust
fn bookmarks_sidecar_path(file_path: &std::path::Path) -> Option<std::path::PathBuf> {
    let dir = file_path.parent()?;
    let name = file_path.file_name()?.to_str()?;
    Some(dir.join(format!(".{name}.hexenly.json")))
}

fn load_bookmarks(file_path: &std::path::Path) -> Vec<Bookmark> {
    let Some(sidecar) = bookmarks_sidecar_path(file_path) else { return vec![] };
    let Ok(content) = std::fs::read_to_string(&sidecar) else { return vec![] };

    #[derive(serde::Deserialize)]
    struct Sidecar { bookmarks: Vec<Bookmark> }

    serde_json::from_str::<Sidecar>(&content)
        .map(|s| s.bookmarks)
        .unwrap_or_default()
}

fn save_bookmarks(file_path: &std::path::Path, bookmarks: &[Bookmark]) {
    let Some(sidecar) = bookmarks_sidecar_path(file_path) else { return };

    #[derive(serde::Serialize)]
    struct Sidecar<'a> { bookmarks: &'a [Bookmark] }

    let json = serde_json::to_string_pretty(&Sidecar { bookmarks }).unwrap_or_default();
    if let Err(e) = std::fs::write(&sidecar, json) {
        tracing::error!("Failed to save bookmarks: {e}");
    }
}
```

```rust
use egui::{RichText, ScrollArea, Ui};
use hexenly_core::Bookmark;

#[derive(Debug)]
pub enum BookmarkAction {
    Add,
    GoToOffset(usize),
    Delete(usize),
    UpdateNote { index: usize, note: String },
    UpdateName { index: usize, name: String },
}

pub fn show(
    ui: &mut Ui,
    bookmarks: &mut Vec<Bookmark>,
    cursor_offset: usize,
) -> Option<BookmarkAction> {
    let mut action = None;

    ui.heading("Bookmarks");
    ui.separator();

    if ui.button("+ Add at cursor").clicked() {
        action = Some(BookmarkAction::Add);
    }

    ui.add_space(4.0);

    ScrollArea::vertical()
        .auto_shrink([false, false])
        .show(ui, |ui| {
            let mut delete_idx = None;
            for (idx, bookmark) in bookmarks.iter_mut().enumerate() {
                ui.group(|ui| {
                    ui.horizontal(|ui| {
                        // Clickable name to jump
                        let name_response = ui.add(
                            egui::TextEdit::singleline(&mut bookmark.name)
                                .desired_width(100.0)
                                .font(egui::TextStyle::Body),
                        );
                        if name_response.lost_focus() {
                            action = Some(BookmarkAction::UpdateName {
                                index: idx,
                                name: bookmark.name.clone(),
                            });
                        }

                        // Offset label — click to jump
                        let offset_text = format!("0x{:08X}", bookmark.offset);
                        if ui
                            .add(egui::Label::new(
                                RichText::new(&offset_text).monospace().weak(),
                            ).sense(egui::Sense::click()))
                            .clicked()
                        {
                            action = Some(BookmarkAction::GoToOffset(bookmark.offset));
                        }

                        // Delete button
                        if ui.small_button("X").clicked() {
                            delete_idx = Some(idx);
                        }
                    });

                    // Note field
                    let note_response = ui.add(
                        egui::TextEdit::singleline(&mut bookmark.note)
                            .hint_text("Add a note...")
                            .desired_width(ui.available_width())
                            .font(egui::TextStyle::Small),
                    );
                    if note_response.lost_focus() {
                        action = Some(BookmarkAction::UpdateNote {
                            index: idx,
                            note: bookmark.note.clone(),
                        });
                    }
                });
                ui.add_space(2.0);
            }

            if let Some(idx) = delete_idx {
                action = Some(BookmarkAction::Delete(idx));
            }
        });

    if bookmarks.is_empty() {
        ui.label("No bookmarks. Press Ctrl+B or click + to add.");
    }

    action
}
```

**Step 1:** Create `.github/workflows/ci.yml`:

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  CARGO_TERM_COLOR: always

jobs:
  build:
    strategy:
      matrix:
        os: [ubuntu-latest, macos-latest, windows-latest]
    runs-on: ${{ matrix.os }}

    steps:
      - uses: actions/checkout@v4

      - name: Install Rust toolchain
        uses: dtolnay/rust-toolchain@stable

      - name: Cache cargo registry & build
        uses: actions/cache@v4
        with:
          path: |
            ~/.cargo/registry
            ~/.cargo/git
            target
          key: ${{ runner.os }}-cargo-${{ hashFiles('**/Cargo.lock') }}
          restore-keys: ${{ runner.os }}-cargo-

      - name: Install Linux dependencies
        if: runner.os == 'Linux'
        run: sudo apt-get update && sudo apt-get install -y libgtk-3-dev

      - name: Build
        run: cargo build --workspace

      - name: Test
        run: cargo test --workspace

      - name: Clippy
        run: cargo clippy --workspace -- -D warnings
```

**Note:** eframe/egui on Linux requires GTK3 dev headers for the native file dialog (rfd). The `libgtk-3-dev` install step handles this.

### Task 7: Release Workflow

Create GitHub Actions workflow that builds release binaries on version tags and uploads them to GitHub Releases.

**Files:**
- Create: `.github/workflows/release.yml`

**Step 1:** Create `.github/workflows/release.yml`:

```yaml
name: Release

on:
  push:
    tags: ['v*']

permissions:
  contents: write

env:
  CARGO_TERM_COLOR: always

jobs:
  build:
    strategy:
      matrix:
        include:
          - os: ubuntu-latest
            target: x86_64-unknown-linux-gnu
            artifact: hexenly-linux-x86_64
          - os: macos-latest
            target: x86_64-apple-darwin
            artifact: hexenly-macos-x86_64
          - os: windows-latest
            target: x86_64-pc-windows-msvc
            artifact: hexenly-windows-x86_64.exe
    runs-on: ${{ matrix.os }}

    steps:
      - uses: actions/checkout@v4

      - name: Install Rust toolchain
        uses: dtolnay/rust-toolchain@stable
        with:
          targets: ${{ matrix.target }}

      - name: Install Linux dependencies
        if: runner.os == 'Linux'
        run: sudo apt-get update && sudo apt-get install -y libgtk-3-dev

      - name: Build release binary
        run: cargo build --release -p hexenly-app --target ${{ matrix.target }}

      - name: Rename binary (Unix)
        if: runner.os != 'Windows'
        run: cp target/${{ matrix.target }}/release/hexenly-app ${{ matrix.artifact }}

      - name: Rename binary (Windows)
        if: runner.os == 'Windows'
        run: cp target/${{ matrix.target }}/release/hexenly-app.exe ${{ matrix.artifact }}

      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.artifact }}
          path: ${{ matrix.artifact }}

  release:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Download all artifacts
        uses: actions/download-artifact@v4
        with:
          path: artifacts
          merge-multiple: true

      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2
        with:
          files: artifacts/*
          generate_release_notes: true
```
